55  /**
56   * OAuth Wire.
57   *
58   * @author Yegor Bugayenko (yegor@tpc2.com)
59   * @version $Id$
60   * @since 0.1
61   */
62  @Immutable
63  @ToString
64  @EqualsAndHashCode(of = { "origin", "key", "secret", "token", "tsecret" })
65  public final class OAuthWire implements Wire {
66  
67      /**
68       * Original wire.
69       */
70      private final transient Wire origin;
71  
72      /**
73       * Application key.
74       */
75      private final transient String key;
76  
77      /**
78       * Application secret.
79       */
80      private final transient String secret;
81  
82      /**
83       * Access token.
84       */
85      private final transient String token;
86  
87      /**
88       * Access token secret.
89       */
90      private final transient String tsecret;
91  
92      /**
93       * Public ctor.
94       * @param wire Original wire
95       * @param akey App key
96       * @param scrt App secret
97       * @param tkn OAuth access token
98       * @param tscrt OAuth access token secret part
99       * @checkstyle ParameterNumber (10 lines)
100      */
101     public OAuthWire(final Wire wire, final String akey, final String scrt,
102         final String tkn, final String tscrt) {
103         this.origin = wire;
104         this.key = akey;
105         this.secret = scrt;
106         this.token = tkn;
107         this.tsecret = tscrt;
108     }
109 
110     // @checkstyle ParameterNumber (10 lines)
111     @Override
112     public Response send(final Request req, final String home,
113         final String method,
114         final Collection<Map.Entry<String, String>> headers,
115         final InputStream content) throws IOException {
116         final OAuthService service = new ServiceBuilder()
117             .provider(OAuthWire.OdeskApi.class)
118             .apiKey(this.key)
119             .apiSecret(this.secret)
120             .debugStream(Logger.stream(Level.FINE, this))
121             .build();
122         final String[] parts = home.split("\\?", 2);
123         final OAuthRequest oauth = new OAuthRequest(
124             Verb.valueOf(method), parts[0]
125         );
126         if (parts.length == 2) {
127             for (final String pair : parts[1].split("&")) {
128                 final String[] eqn = pair.split("=", 2);
129                 final String value;
130                 if (eqn.length == 2) {
131                     value = URLDecoder.decode(eqn[1], "UTF-8");
132                 } else {
133                     value = "";
134                 }
135                 oauth.addQuerystringParameter(eqn[0], value);
136             }
137         }
138         service.signRequest(new Token(this.token, this.tsecret), oauth);
139         return this.origin.send(
140             req, oauth.getCompleteUrl(), method,
141             new Array<Map.Entry<String, String>>(headers).with(
142                 new AbstractMap.SimpleEntry<String, String>(
143                     HttpHeaders.AUTHORIZATION,
144                     oauth.getHeaders().get(HttpHeaders.AUTHORIZATION)
145                 )
146             ),
147             content
148         );
149     }
150 
151     /**
152      * Odesk provider.
153      * @link https://github.com/fernandezpablo85/scribe-java/pull/438
154      */
155     public static final class OdeskApi extends DefaultApi10a {
156         @Override
157         public String getAccessTokenEndpoint() {
158             return "https://www.odesk.com/api/auth/v1/oauth/token/access";
159         }
160         @Override
161         public String getAuthorizationUrl(final Token tkn) {
162             return String.format(
163                 "https://www.odesk.com/services/api/auth?oauth_token=%s",
164                 tkn.getToken()
165             );
166         }
167         @Override
168         public String getRequestTokenEndpoint() {
169             return "https://www.odesk.com/api/auth/v1/oauth/token/request";
170         }
171     }
